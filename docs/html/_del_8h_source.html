123 fir_gen(const std::vector<S> &ir) {
124  return [&](S rp, std::size_t wp, const std::vector<S> &del) -> S {
125  auto dl = del.begin() + wp;
126  S mx = 0;
127  for (auto irs = ir.rbegin(); irs != ir.rend(); irs++, dl++) {
128  if (dl == del.end())
129  dl = del.begin();
130  mx += *dl * *irs;
131  }
132  return mx;
133  };

147  S delay(S in, S dt, S fdb, S fwd, std::size_t &p,
148  std::function<S(S, std::size_t, const std::vector<S> &)> fn) {
149  S s = fn(dt * fs, p, del);
150  S w = in + s * fdb;
151  del[p] = w;
152  p = p < del.size() - 1 ? p + 1 : 0;
153  return w * fwd + s;
154  }
