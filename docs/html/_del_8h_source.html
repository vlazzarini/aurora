54template <typename S> inline S rpos(S rp, std::size_t wp, std::size_t ds) {
55 rp = wp - rp;
56 while (rp < 0)
57 rp += ds;
58 while (rp >= ds)
59 rp -= ds;
60 return rp;
61}

121template <typename S>
122inline S lp_delay(S nop, std::size_t wp, const std::vector<S> &d,
123 std::vector<S> *mem) {
124 S ym1 = (*mem)[0];
125 S coef = (*mem)[1];
126 S x = d[wp];
127 S y = (1 + coef) * x - coef * ym1;
128 (*mem)[0] = y;
129 return y;
130}

139template <typename S>
140inline S fir(S nop, std::size_t wp, const std::vector<S> &del,
141 std::vector<S> *ir) {
142 auto dl = del.begin() + wp;
143 S mx = 0;
144 for (auto irs = ir->rbegin(); irs != ir->rend(); irs++, dl++) {
145 if (dl == del.end())
146 dl = del.begin();
147 mx += *dl * *irs;
148 }
149 return mx;
150};

156template <typename S, S (*FN)(S, std::size_t, const std::vector<S> &,
157 std::vector<S> *) = fixed_delay>
158class Del : public SndBase<S> {
159 using SndBase<S>::process;
160 S fs;
161 std::size_t wp;
162 std::vector<S> del;
163
164 S delay(S in, S dt, S fdb, S fwd, std::size_t &p, std::vector<S> *mem) {
165 S s = FN(dt, p, del, mem);
166 S w = in + s * fdb;
167 del[p] = w;
168 p = p < del.size() - 1 ? p + 1 : 0;
169 return w * fwd + s;
170 }
