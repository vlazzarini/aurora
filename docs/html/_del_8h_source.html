58  std::size_t ds = del.size();

113  S delay(S in, S dt, S fdb, S fwd, std::size_t &p) {
114  S s = fun(dt * fs, p, del);
115  S w = in + s * fdb;
116  del[p] = w;
117  p = p < del.size() - 1 ? p + 1 : 0;
118  return w * fwd + s;
119  }
