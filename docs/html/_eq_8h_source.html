35template <typename S> class Eq : SndBase<S> {
36 using SndBase<S>::process;
37 double z[2];
38 double d, a;
39 double piosr;
40 S ff, bbw;
41
42 S filter(S in, S g, double *z, double d, double a) {
43 double w = in + d * (1.0 + a) * z[0] - a * z[1];
44 double y = w * a - d * (1.0 + a) * z[0] + z[1];
45 z[1] = z[0];
46 z[0] = w;
47 return (S)(0.5 * (y + in + g * (in - y)));
48 }
49
50 void coeffs(S f, S bw) {
51 S c = tan(piosr * bw);
52 d = std::cos(2 * piosr * f);
53 a = (1. - c) / (1. + c);
54 ff = f;
55 bbw = bw;
56 }
57
58public:
63 Eq(S fs = def_sr, std::size_t vsize = def_vsize)
64 : SndBase<S>(vsize), z{0., 0.}, d(0.), a(0.), piosr(M_PI / fs), ff(0),
65 bbw(0){};
66
73 const std::vector<S> &operator()(const std::vector<S> &in, S g, S fr, S bw) {
74 std::size_t n = 0;
75 double dd = d, aa = a;
76 if (ff != fr || bw != bbw)
77 coeffs(fr, bw);
78 return process([&]() { return filter(in[n++], g, z, dd, aa); }, in.size());
79 }
80
84 void reset(S fs) {
85 piosr = M_PI / fs;
86 coeffs(ff, bbw);
87 }
88};
