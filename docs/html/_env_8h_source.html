46template <typename S> inline S ads(S a, S d, S s, double t, S e, S ts) {
47 if (t < a && e < 1)
48 return e + ts / a;
49 else if (t < a + d && e > s && d != 0)
50 return e + (s - 1) * ts / d;
51 else
52 return s;
53}

60template <typename S>
61std::function<S(double, S, S)> env_gen(const std::vector<S> &pts) {
62 return [&pts](double t, S e, S ts) -> S {
63 std::size_t n, siz = pts.size() % 2 ? pts.size() - 1 : pts.size();
64 for (n = 0; n < siz; n += 2) {
65 if (t < pts[n]) {
66 S p1 = n ? pts[n - 1] : 0;
67 S p2 = pts[n + 1];
68 S st = n ? pts[n - 2] : 0;
69 return e + (p2 - p1) * ts / (pts[n] - st);
70 }
71 }
72 return *(pts.end() - 1);
73 };
74}

106 S synth(S e, double &t, bool gate, std::function<S(double, S, S)> fn, S a,
107 S b, S c) {
108 S s;
109 if (gate) {
110 s = fn ? fn(t, e, ts) : FN(a, b, c, t, e, ts);
111 t += ts;
112 } else {
113 if (e < 0.00001)
114 s = 0;
115 else
116 s = e * fac;
117 t = 0;
118 }
119 return s;
120 }
