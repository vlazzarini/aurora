46 template <typename S> inline S ads(S a, S d, S s, double t, S e, S ts) {
47  if (t < a && e < 1)
48  return e + ts / a;
49  else if (t < a + d && e > s)
50  return e + (s - 1) * ts / d;
51  else
52  return s;
53 }

60 template <typename S>
61 std::function<S(double, S, S)> env_gen(const std::vector<S> &pts) {
62  return [&pts](double t, S e, S ts) -> S {
63  std::size_t n, siz = pts.size() % 2 ? pts.size() - 1 : pts.size();
64  for (n = 0; n < siz; n += 2) {
65  if (t < pts[n]) {
66  S p1 = n ? pts[n - 1] : 0;
67  S p2 = pts[n + 1];
68  S st = n ? pts[n - 2] : 0;
69  return e + (p2 - p1) * ts / (pts[n] - st);
70  }
71  }
72  return *(pts.end() - 1);
73  };
74 }

105  S synth(S e, double &t, bool gate, std::function<S(double, S, S)> fn, S a,
106  S d, S ss) {
107  S s;
108  if (gate) {
109  s = fn ? fn(t, e, ts) : FN(a, d, ss, t, e, ts);
110  t += ts;
111  } else {
112  if (e < 0.00001)
113  s = 0;
114  else
115  s = e * fac;
116  t = 0;
117  }
118  return s;
119  }
