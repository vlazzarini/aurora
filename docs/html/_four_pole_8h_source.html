38 template <typename S> class FourPole : public SndBase<S> {
39  using SndBase<S>::sig;
40  double D[4];
41  double A, G[4];
42  S ff;
43  double piosr;
44 
45  S filter(S s, double *d, double *g, double a, S k) {
46  S o, u, w;
47  S ss = d[3];
48  for (int j = 0; j < 3; j++)
49  ss += d[j] * g[2 - j];
50  o = (g[3] * s + ss) / (1 - k * g[3]);
51  u = g[0] * (s - k * o);
52  for (int j = 0; j < 3; j++) {
53  w = d[j] + u;
54  d[j] = u - a * w;
55  u = g[0] * w;
56  }
57  d[3] = g[0] * w - a * o;
58  return o;
59  }
60 
61  void coeffs(S f) {
62  S g;
63  ff = f;
64  g = std::tan(f * piosr);
65  G[0] = g / (1 + g);
66  A = A = (g - 1) / (1 + g);
67  G[1] = G[0] * G[0];
68  G[2] = G[0] * G[1];
69  G[3] = G[0] * G[2];
70  }
