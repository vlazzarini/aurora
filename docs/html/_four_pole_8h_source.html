40template <typename S> class FourPole : public SndBase<S> {
41 using SndBase<S>::process;
42 double D[4];
43 double A, G[4];
44 S ff;

47 S filter(S s, double *d, double *g, double a, S k) {
48 S o, u, w;
49 S ss = d[3];
50 for (int j = 0; j < 3; j++)
51 ss += d[j] * g[2 - j];
52 o = (g[3] * s + ss) / (1 + k * g[3]);
53 u = g[0] * (s - k * o);
54 for (int j = 0; j < 3; j++) {
55 w = d[j] + u;
56 d[j] = u - a * w;
57 u = g[0] * w;
58 }
59 d[3] = g[0] * w - a * o;
60 return o;
61 }
62
63 void coeffs(S f, double *g, double &a, S &of, double ts) {
64 S w = std::tan(f * ts);
65 g[0] = w / (1 + w);
66 a = (w - 1) / (1 + w);
67 g[1] = g[0] * g[0];
68 g[2] = g[0] * g[1];
69 g[3] = g[0] * g[2];
70 of = f;
71 }
