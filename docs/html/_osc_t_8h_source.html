112  virtual S synth(S a, S f, double &phs, const std::vector<S> *t) {
113  while (phs < 0)
114  phs += 1.;
115  while (phs >= 1.)
116  phs -= 1.;
117  S s = (S)(a * FN(phs, t));
118  phs += f * ts;
119  return s;
120  }
