39template <typename S> class OnePole : public SndBase<S> {
40 using SndBase<S>::process;
41 double D;
42 double A, G;
43 S ff;
44 double piosr;
45
46 S filter(S s, double &d, double g, double a) {
47 S u = g * s;
48 S y = u + d;
49 d = u - a * y;
50 return y;
51 }
52
53 void coeffs(S f, double &g, double &a, S &of, double ts) {
54 S w = std::tan(f * ts);
55 g = w / (1 + w);
56 a = (w - 1) / (1 + w);
57 of = f;
58 }

60public:
65 OnePole(S fs = def_sr, int vsize = def_vsize)
66 : SndBase<S>(vsize), D(0), A(0), G(0), ff(0), piosr(M_PI / fs){};
67
72 const std::vector<S> &operator()(const std::vector<S> &in, S f) {
73 double d = D;
74 std::size_t n = 0;
75 if (f != ff)
76 coeffs(f, G, A, ff, piosr);
77 auto &s = process([&]() { return filter(in[n++], d, G, A); }, in.size());
78 D = d;
79 return s;
80 }

86 const std::vector<S> &operator()(const std::vector<S> &in,
87 const std::vector<S> &f) {
88 double d = D;
89 std::size_t n = 0;
90 auto &s = process(
91 [&]() {
92 if (f[n] != ff)
93 coeffs(f[n], G, A, ff, piosr);
94 return filter(in[n++], d, G, A);
95 },
96 in.size() < f.size() ? in.size() : f.size());
97 D = d;
98 return s;
99 }

104 void reset(S fs) {
105 D = 0;
106 piosr = M_PI / fs;
107 coeffs(ff, G, A, ff, piosr);
108 }
