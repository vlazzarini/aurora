37 template <typename S> void reson(S in, double *c, double *d) {
38  S y = in * c[0] - d[0] * c[0] - d[1] * c[1];
39  c[1] = c[0];

46 template <typename S> void reson_coeffs(S fr, S bw, S fs, double *c) {
47  double r = std::exp(-bw * M_PI / fs);
48  double rr = 2. * r;
49  double rsq = r * r;
50  costh = (rr / (1. + rsq)) * std::cos(twopi * f / fs);
51  c[0] = (1 - rsq) * std::sin(std::acos(costh));
52  c[1] = -rr * costh;
53  c[2] = rsq;
54 }
