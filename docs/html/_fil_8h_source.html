37template <typename S> inline S reson(S in, double *c, double *d) {
38 S y = in * c[0] - d[0] * c[1] - d[1] * c[2];
39 d[1] = d[0];
40 d[0] = y;
41 return y;
42}

47template <typename S> inline void reson_cfs(S f, S bw, S fs, double *c) {
48 c[2] = std::exp(-bw * twopi / fs);
49 c[1] = (-4 * c[2] / (1. + c[2])) * std::cos(twopi * f / fs);
50}

55template <typename S> inline void reson_cfs1(S f, S bw, S fs, double *c) {
56 reson_cfs(f, bw, fs, c);

63template <typename S> inline void reson_cfs2(S f, S bw, S fs, double *c) {
64 reson_cfs(f, bw, fs, c);

71template <typename S> inline S dfI(S in, double *c, double *d) {

92template <typename S> inline void lp_cfs(S f, S bw, S fs, double *c) {
93 double w = 1 / tan(M_PI * f / fs);
94 double sqw = sqrt(2.) * w;
95 double wsq = w * w;
96 c[0] = 1 / (1 + sqw + wsq);
97 c[1] = 2 * c[0];
98 c[2] = c[0];
99 c[3] = 2 * (1. - sqw) * c[0];
100 c[4] = (1 - sqw + wsq) * c[0];
101}

105template <typename S> inline void hp_cfs(S f, S bw, S fs, double *c) {
106 double w = tan(M_PI * f / fs);
107 double sqw = sqrt(2.) * w;
108 double wsq = w * w;
109 c[0] = 1 / (1 + sqw + wsq);
110 c[1] = -2 * c[0];
111 c[2] = c[0];
112 c[3] = 2 * (wsq - 1) * c[0];
113 c[4] = (1 - sqw + wsq) * c[0];
114}

118template <typename S> inline void bp_cfs(S f, S bw, S fs, double *c) {
119 double w = 1. / tan(M_PI * bw / fs);
120 double cosw = 2. * cos(2 * M_PI * f / fs);
121 c[0] = 1. / (1 + w);
122 c[1] = 0;
123 c[2] = -c[0];
124 c[3] = -w * cosw * c[0];
125 c[4] = (w - 1) * c[0];
126}

130template <typename S> inline void br_cfs(S f, S bw, S fs, double *c) {
131 double w = tan(M_PI * bw / fs);
132 double cosw = 2. * cos(2 * M_PI * f / fs);
133 c[0] = 1 / (1 + w);
134 c[1] = -cosw * c[0];
135 c[2] = c[0];
136 c[3] = c[1];
137 c[4] = (1 - w) * c[0];
138}

175 const std::vector<S> &operator()(const std::vector<S> &in, S f, S bw = 0) {
176 std::size_t n = 0;
177 double *D = d, *C = c;
178 if (f != ff || bw != bbw)
179 coeffs(f, bw, fs, c);
180 auto &s = process([&]() { return filter(in[n++], C, D); }, in.size());
181 return s;
182 }

189 const std::vector<S> &operator()(const std::vector<S> &in,
190 const std::vector<S> &f, S bw = 0) {
191 std::size_t n = 0;
192 double *D = d, *C = c;
193 auto &s = process(
194 [&]() {
195 if (f[n] != ff)
196 coeffs(f, bw, fs, C);
197 return filter(in[n++], C, D);
198 },
199 in.size() < f.size() ? in.size() : f.size());
200 return s;
201 }

207 d[0] = d[1] = d[2] = d[3] = 0;
208 coeffs(ff, bbw, fs, c, d);

210 }
