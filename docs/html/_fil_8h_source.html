92 template <typename S> inline void lp_cfs(S f, S bw, S fs, double *c) {
93  double w = 1 / tan(M_PI * f / fs);
94  double sqw = sqrt(2.) * w;
95  double wsq = w * w;
96  c[0] = 1 / (1 + sqw + wsq);
97  c[1] = 2 * c[0];
98  c[2] = c[0];
99  c[3] = 2 * (1. - sqw) * c[0];
100  c[4] = (1 - sqw + wsq) * c[0];
101 }

118 template <typename S> inline void bp_cfs(S f, S bw, S fs, double *c) {
119  double w = 1. / tan(M_PI * bw / fs);
120  double cosw = 2. * cos(2 * M_PI * f / fs);
121  c[0] = 1. / (1 + w);
122  c[1] = 0;
123  c[2] = -c[0];
124  c[3] = -w * cosw * c[0];
125  c[4] = (w - 1) * c[0];

130 template <typename S> inline void br_cfs(S f, S bw, S fs, double *c) {
131  double w = tan(M_PI * bw / fs);
132  double cosw = 2. * cos(2 * M_PI * f / fs);
