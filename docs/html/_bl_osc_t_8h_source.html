49  void norm(std::vector<S> &wave) {
50  S max = 0.;
51  for (auto s : wave)
52  if (s > max)
53  max = s;
54  for (auto &s : wave)
55  s *= 1. / max;
56  }

58  void fourier(const std::vector<S> &src, S fs, int32_t type = -1) {
59  std::size_t nh;
60  uint32_t k = 0;
61  FFT<S> fft(tlen);
62  std::vector<std::complex<S>> blsp(fft.size() / 2);
63  if (type < 0) {
64  auto sp = fft.transform(src);
65  std::copy(sp, sp + blsp.size(), blsp.begin());
66  } else {
67  std::size_t n = 0;
68  for (auto &s : blsp) {
69  if (type > 1)
70  s.real(src[n++]);
71  else
72  s.imag(-src[n++]);
73  }
74  }
75  for (auto &wave : waves) {
76  double fr = base * std::pow(2, k++);
77  if (fr > fs * .375)
78  nh = 2;
79  else
80  nh = .375 * fs / fr + 1;
81  std::fill(blsp.begin() + nh, blsp.end(), std::complex<S>(0, 0));
82  auto wv = fft.transform(blsp);
83  std::copy(wv, wv + tlen, wave.begin());
84  norm(wave);
85  }
86  }

88  void create(S fs, uint32_t type) {
89  std::vector<S> src(tlen / 2);
90  if (type >= 0) {
91  std::size_t n = 0;
92  for (auto &s : src) {
93  switch (type) {
94  case SAW:
95  if (n)
96  s = (S)1. / n;
97  break;
98  case SQUARE:
99  if (n % 2)
100  s = (S)1. / n;
101  break;
102  case TRIANGLE:
103  if (n % 2)
104  s = (S)1. / (n * n);
105  break;
106  default:
107  s = (S)1.;
108  }
109  n++;
110  }
111  }
112  fourier(src, fs, type);
113  }
