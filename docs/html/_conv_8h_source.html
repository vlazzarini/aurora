40 template <typename S> class IR {
41  std::vector<std::vector<std::complex<S>>> parts;

43  void create(const std::vector<S> &s, std::size_t psize) {
44  std::vector<S> buffer(psize);
45  std::size_t n = 0, k = 0, cnt = s.size();
46  FFT<S> fft(2 * psize, !packed, inverse);
47 
48  while (cnt) {
49  cnt = cnt > psize ? psize : cnt;
50  std::copy(s.begin() + n, s.begin() + n + cnt, buffer.begin());
51  fft.transform(buffer);
52  auto v = fft.vector();
53  std::copy(v.begin(), v.end(), parts[k++].begin());
54  std::fill(buffer.begin(), buffer.end(), 0);
55  n += cnt;
56  cnt = s.size() - n;
57  }
58  }

106 template <typename S> class Conv : public SndBase<S> {
107  using SndBase<S>::process;

114  FFT<S> fft;

117  void convol(const std::vector<S> &in) {
118  fft.transform(in);
119  auto &v = fft.vector();
120  std::copy(v.begin(), v.end(), del[p].begin());
121  std::fill(mix.begin(), mix.end(), 0);
122  p = p == del.size() - 1 ? 0 : p + 1;
123  auto dl = del.begin() + p;
124  auto &part = ir->spectrum();
125  for (auto prt = part.rbegin(); prt != part.rend(); prt++, dl++) {
126  if (dl == del.end())
127  dl = del.begin();
128  auto dsamp = dl->begin();
129  auto psamp = prt->begin();
130  for (auto &mx : mix)
131  mx += (*dsamp++ * *psamp++);
132  }
133  fft.transform(mix);
134  }

137  std::size_t psz) {
138  auto s = bufout[cnt] + olabuf[cnt];
139  bufin[cnt] = in;
140  olabuf[cnt] = bufout[cnt + psz];
141  return s;
142  }
