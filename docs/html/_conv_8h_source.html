42  void create(const std::vector<S> &s, std::size_t psize) {
43  std::vector<S> buffer(psize);
44  std::size_t n = 0, k = 0, cnt = s.size();
45  FFT<S> fft(2 * psize, !packed, inverse);
46 
47  while (cnt) {
48  cnt = cnt > psize ? psize : cnt;
49  std::copy(s.begin() + n, s.begin() + n + cnt, buffer.begin());
50  fft.transform(buffer);
51  auto v = fft.vector();
52  std::copy(v.begin(), v.end(), parts[k++].begin());
53  std::fill(buffer.begin(), buffer.end(), 0);
54  n += cnt;
55  cnt = s.size() - n;
56  }
57  }

142 template <typename S> class Conv : public SndBase<S> {
143  using SndBase<S>::process;
144  const IR<S> *ir;
145  std::vector<std::vector<std::complex<S>>> del;
146  std::vector<std::complex<S>> mix;
147  std::vector<S> inbuf;
148  std::vector<S> olabuf;
149  std::size_t p, sn;
150  FFT<S> fft;
151  std::function<S(S, S *, const S *, S *, std::size_t, std::size_t)> fun;
152 
153  void convol(const std::vector<S> &in) {
154  fft.transform(in);
155  auto &v = fft.vector();
156  std::copy(v.begin(), v.end(), del[p].begin());
157  std::fill(mix.begin(), mix.end(), 0);
158  p = p == del.size() - 1 ? 0 : p + 1;
159  auto dl = del.begin() + p;
160  auto &part = ir->spectrum();
161  for (auto prt = part.rbegin(); prt != part.rend(); prt++, dl++) {
162  if (dl == del.end())
163  dl = del.begin();
164  auto dsamp = dl->begin();
165  auto psamp = prt->begin();
166  for (auto &mx : mix)
167  mx += (*dsamp++ * *psamp++);
168  }
169  fft.transform(mix);
170  }
171

180  f = ols<S>,

182  : SndBase<S>(vsize), ir(imp),

191  const std::vector<S> &operator()(const std::vector<S> &in, S scal) {
