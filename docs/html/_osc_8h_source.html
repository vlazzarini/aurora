52 template <typename S> std::function<S(S)> lookupi_gen(const std::vector<S> &t) {
53  return [&t](double ph) -> S {
54  double pos = ph * t.size();
55  size_t posi = (size_t)pos;
56  double frac = pos - posi;
57  return t[posi] +
58  frac * ((posi != t.size() - 1 ? t[posi + 1] : t[0]) - t[posi]);
59  };
60 }
