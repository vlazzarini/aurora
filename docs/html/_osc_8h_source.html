98  virtual S synth(S a, S f, double &phs, std::function<S(S)> fn) {
99  while (phs < 0)
100  phs += 1.;
101  while (phs >= 1.)
102  phs -= 1.;
103  S s = (S)(a * fn(phs));
104  phs += f * ts;
105  return s;
106  }
