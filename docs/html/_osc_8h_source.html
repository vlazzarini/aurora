50 template <typename S> std::function<S(S)> lookupi_gen(const std::vector<S> &t) {
51  return [&t](double ph) -> S {
52  double pos = ph * t.size();
53  size_t posi = (size_t)pos;
54  double frac = pos - posi;
55  return t[posi] +
56  frac * ((posi != t.size() - 1 ? t[posi + 1] : t[0]) - t[posi]);
57  };
58 }
