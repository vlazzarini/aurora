103template <typename S, S (*FN)(double, const std::vector<S> *) = cos>
104class Osc : public SndBase<S> {
105 using SndBase<S>::process;
106
107protected:
108 double ph;
109 S ts;
110 const std::vector<S> *tab;
111
112 virtual S synth(S a, S f, double &phs, const std::vector<S> *t, S pm = 0) {
113 phs += pm;
114 while (phs < 0)
115 phs += 1.;
116 while (phs >= 1.)
117 phs -= 1.;
118 S s = (S)(a * FN(phs, t));
119 phs = f * ts + phs - pm;
120 return s;
121 }
122
123 virtual void reset_obj(S fs) {
124 ts = 1 / fs;
125 ph = 0.;
126 }
127
128public:
135 Osc(const std::vector<S> *t, S fs = (S)def_sr, std::size_t vsize = def_vsize)
136 : SndBase<S>(vsize), ph(0.), ts(1 / fs), tab(t){};
