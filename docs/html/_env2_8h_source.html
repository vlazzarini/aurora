43 template <typename S>
44 std::function<S(double, S, S)> ads_gen(const S &a, const S &d, const S &s) {
45  return [&a, &d, &s](double t, S e, S ts) -> S {
46  if (t < a)
47  return e + ts / a;
48  else if (t < a + d)
49  return e + (s - 1) * ts / d;
50  else
51  return s;
52  };
53 }

67  S synth(S e, double &t, bool gate, std::function<S(double, S, S)> fn) {
68  S s;
69  if (gate) {
70  s = fn(t, e, ts);
71  t += ts;
72  } else {
73  if (e < 0.00001)
74  s = 0;
75  else
76  s = e * fac;
77  t = 0;
78  }
79  return s;
80  }
