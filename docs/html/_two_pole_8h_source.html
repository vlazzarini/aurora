46template <typename S, S (*FN)(S, S) = id> class TwoPole : public SndBase<S> {
47 using SndBase<S>::process;
48 S Y[2];
49 double D[2];
50 double W, Fac;

53
54 S filter(S in, S *y, double *s, double w, double fac, S d, S drv, int32_t typ,
55 S m) {
56 S lp;
57 y[HP] = (in - (d + w) * s[0] - s[1]) * fac;
58 S u = w * FN(y[0], drv);
59 y[BP] = u + s[0];
60 s[0] = y[BP] + u;
61 u = w * FN(y[1], drv);
62 lp = u + s[1];
63 s[1] = lp + u;
64 return typ == -1 ? lp * (1 - m) + Y[HP] * m : Y[HP] * (1 - m) + Y[BP] * m;
65 }
66
67 void coeffs(S f, S d, S &od, double &w, double &fac, S &of, double ts) {
68 w = std::tan(f * ts);
69 fac = 1. / (1. + w * d + w * w);
70 of = f;
71 od = d;
72 }
