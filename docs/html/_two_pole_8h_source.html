49  S filter(S in, S *y, double *s, double w, double fac, S d, S drv, int32_t typ,
50  S m) {
51  S lp;
52  y[HP] = (in - (d + w) * s[0] - s[1]) * fac;
53  S u = w * fun(y[0] * drv) * 1 / drv;
54  y[BP] = u + s[0];
55  s[0] = y[BP] + u;
56  u = w * fun(y[1] * drv) * 1 / drv;
57  lp = u + s[1];
58  s[1] = lp + u;
59  return typ == -1 ? lp * (1 - m) + Y[HP] * m : Y[HP] * (1 - m) + Y[BP] * m;
60  }

62  void coeffs(S f, S d, S &od, double &w, double &fac, S &of, double ts) {
63  w = std::tan(f * ts);
64  fac = 1. / (1. + w * d + w * w);
65  of = f;
66  od = d;
67  }
