53  void reorder(std::vector<std::complex<S>> &s) {
54  uint32_t N = s.size();
55  uint32_t j = 0, m;
56  for (uint32_t i = 0; i < N; i++) {
57  if (j > i) {
58  std::swap(s[i], s[j]);
59  }
60  m = N / 2;
61  while (m >= 2 && j >= m) {
62  j -= m;
63  m /= 2;
64  }
65  j += m;
66  }
67  }

85  void transform(std::vector<std::complex<S>> &s, bool dir) {
86  uint32_t N = s.size();
87  std::complex<S> wp, w, even, odd;
88  S o;
89  uint32_t i;
90  reorder(s);
91  for (uint32_t n = 1; n < N; n *= 2) {
92  o = dir == forward ? -M_PI / n : M_PI / n;
93  wp.real(std::cos(o)), wp.imag(std::sin(o));
94  w = 1.;
95  for (uint32_t m = 0; m < n; m++) {
96  for (uint32_t k = m; k < N; k += n * 2) {
97  i = k + n;
98  even = s[k];
99  odd = w * s[i];
100  s[k] = even + odd;
101  s[i] = even - odd;
102  }
103  w *= wp;
104  }
105  }
106  if (dir == forward)
107  for (uint32_t n = 0; n < N; n++)
108  s[n] /= N;
109  }

115  const std::complex<S> *transform(const std::vector<S> &r) {
116  using namespace std::complex_literals;
117  uint32_t N = c.size() - (pckd ? 0 : 1);
118  std::complex<S> wp, w = 1., even, odd;
119  S o, zro, nyq;
120  S *s = reinterpret_cast<S *>(c.data());
121  std::fill(c.begin(), c.end(), std::complex<S>(0, 0));
122  std::copy(r.begin(), r.end(), s);
123  if (!pckd)
124  c.resize(N);
125  transform(c, forward);
126  zro = c[0].real() + c[0].imag();
127  nyq = c[0].real() - c[0].imag();
128  c[0].real(zro * .5), c[0].imag(nyq * .5);
129  o = -M_PI / N;
130  wp.real(std::cos(o)), wp.imag(std::sin(o));
131  w *= wp;
132  for (uint32_t i = 1, j = 0; i < N / 2; i++) {
133  j = N - i;
134  even = S(.5) * (c[i] + conj(c[j]));
135  odd = std::complex<S>(.5i) * (conj(c[j]) - c[i]);
136  c[i] = even + w * odd;
137  c[j] = conj(even - w * odd);
138  w *= wp;
139  }
140  if (!pckd) {
141  c.resize(N + 1);
142  c[N].real(c[0].imag());
143  c[0].imag(0.);
144  c[N].imag(0.);
145  }
