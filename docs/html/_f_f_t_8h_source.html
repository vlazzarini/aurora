74 void reorder(std::complex<S> *s) {
75 const std::size_t N = sz;
76 for (std::size_t i = 0, j = 0, m = 0; i < N; i++) {
77 if (j > i) {
78 std::swap(s[i], s[j]);
79 }
80 m = N / 2;
81 while (m >= 2 && j >= m) {
82 j -= m;
83 m /= 2;
84 }
85 j += m;
86 }
87 }

105 void transform(std::complex<S> *s, bool dir) {
106 const std::size_t N = sz;
107 std::complex<S> wp, w, even, odd;
108 reorder(s);
109 for (std::size_t n = 1, i = 0; n < N; n *= 2) {
110 S o = dir == forward ? -M_PI / n : M_PI / n;
111 wp.real(std::cos(o)), wp.imag(std::sin(o));
112 w = 1.;
113 for (std::size_t m = 0; m < n; m++) {
114 for (std::size_t k = m; k < N; k += n * 2) {
115 i = k + n;
116 even = s[k];
117 odd = w * s[i];
118 s[k] = even + odd;
119 s[i] = even - odd;
120 }
121 w *= wp;
122 }
123 }
124 dir = norm ? dir : !dir;
125 if (dir == forward) {
126 for (std::size_t n = 0; n < N; n++)
127 s[n] /= N;
128 }
129 }

135 const std::complex<S> *transform(const std::vector<S> &r) {
136 using namespace std::complex_literals;
137 const std::size_t N = sz;
138 std::complex<S> wp, w = 1., even, odd;
139 S o, zro, nyq;
140 S *s = reinterpret_cast<S *>(c.data());
141 std::fill(c.begin(), c.end(), std::complex<S>(0, 0));
142 std::copy(r.begin(), r.end(), s);
143 transform(c.data(), forward);
144 zro = c[0].real() + c[0].imag();
145 nyq = c[0].real() - c[0].imag();
146 c[0].real(zro), c[0].imag(nyq);
147 o = -M_PI / N;
148 wp.real(std::cos(o)), wp.imag(std::sin(o));
149 w *= wp;
150 for (std::size_t i = 1, j = 0; i < N / 2; i++) {
151 j = N - i;
152 even = S(.5) * (c[i] + conj(c[j]));
153 odd = std::complex<S>(.5i) * (conj(c[j]) - c[i]);
154 c[i] = even + w * odd;
155 c[j] = conj(even - w * odd);
156 w *= wp;
157 }
158 if (!pckd) {
159 c[N].real(c[0].imag());
160 c[0].imag(0.);
161 c[N].imag(0.);
162 }
163 return c.data();
164 }

170 const S *transform(const std::vector<std::complex<S>> &sp) {
171 using namespace std::complex_literals;
172 const std::size_t N = sz;
173 std::fill(c.begin(), c.end(), std::complex<S>(0, 0));
174 std::complex<S> wp, w = 1., even, odd;
175 S o, zro, nyq;
176 S *s = reinterpret_cast<S *>(c.data());
177 std::copy(sp.begin(), sp.end(), c.begin());
178 if (pckd)
179 zro = c[0].real()*0.5, nyq = c[0].imag()*0.5;
180 else {
181 zro = c[0].real()*0.5;
182 nyq = c[N].real()*0.5;
183 }
184 c[0].real(zro + nyq), c[0].imag(zro - nyq);
185 o = M_PI / N;
186 wp.real(std::cos(o)), wp.imag(std::sin(o));
187 w *= wp;
188 for (std::size_t i = 1, j = 0; i < N / 2; i++) {
189 j = N - i;
190 even = S(.5) * (c[i] + conj(c[j]));
191 odd = std::complex<S>(.5i) * (c[i] - conj(c[j]));
192 c[i] = even + w * odd;
193 c[j] = conj(even - w * odd);
194 w *= wp;
195 }
196 transform(c.data(), inverse);
197 return s;
198 }
