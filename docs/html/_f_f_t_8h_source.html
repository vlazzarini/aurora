68  void reorder(std::vector<std::complex<S>> &s) {
69  uint32_t N = s.size();
70  uint32_t j = 0, m;
71  for (uint32_t i = 0; i < N; i++) {
72  if (j > i) {
73  std::swap(s[i], s[j]);
74  }
75  m = N / 2;
76  while (m >= 2 && j >= m) {
77  j -= m;
78  m /= 2;
79  }
80  j += m;
81  }
82  }

102  void transform(std::vector<std::complex<S>> &s, bool dir) {
103  uint32_t N = s.size();
104  std::complex<S> wp, w, even, odd;
105  S o;
106  uint32_t i;
107  reorder(s);
108  for (uint32_t n = 1; n < N; n *= 2) {
109  o = dir == forward ? -M_PI / n : M_PI / n;
110  wp.real(std::cos(o)), wp.imag(std::sin(o));
111  w = 1.;
112  for (uint32_t m = 0; m < n; m++) {
113  for (uint32_t k = m; k < N; k += n * 2) {
114  i = k + n;
115  even = s[k];
116  odd = w * s[i];
117  s[k] = even + odd;
118  s[i] = even - odd;
119  }
120  w *= wp;
121  }
122  }
123  dir = norm ? dir : !dir;
124  if (dir == forward)
125  for (uint32_t n = 0; n < N; n++)
126  s[n] /= N;
127  }

136  std::complex<S> wp, w = 1., even, odd;

138  S *s = reinterpret_cast<S *>(c.data());

140  std::copy(r.begin(), r.end(), s);
141  if (!pckd)

144  zro = c[0].real() + c[0].imag();
145  nyq = c[0].real() - c[0].imag();

147  o = -M_PI / N;
148  wp.real(std::cos(o)), wp.imag(std::sin(o));
149  w *= wp;

151  j = N - i;
