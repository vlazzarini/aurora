288 inline S cubic_interp(double pos, const std::vector<S> &t) {
289  size_t posi = (size_t)pos;
290  double frac = pos - posi;
291  double a = posi == 0 ? t[t.size() - 1] : t[posi - 1];
292  double b = t[posi];
293  double c = posi != t.size() - 1 ? t[posi + 1] : t[0];
294  double d =
295  posi != t.size() - 2 ? (posi != t.size() - 1 ? t[posi + 2] : t[1]) : t[0];
296  double tmp = d + 3.f * b;
297  double fracsq = frac * frac;
298  double fracb = frac * fracsq;
299  return fracb * (-a - 3 * c + tmp) / 6 + fracsq * ((a + c) / 2 - b) +
300  frac * (c + (-2 * a - tmp) / 6) + b;
301 }

322 inline S cubic_interp_lim(double pos, const std::vector<S> &t) {
323  pos = pos < 0 ? 0 : (pos < t.size() ? pos : t.size() - 1);
324  size_t posi = (size_t)pos;
325  if (posi > 0 && posi < t.size() - 2) {
326  double frac = pos - posi;
327  double a = t[posi - 1];
328  double b = t[posi];
329  double c = t[posi + 1];
330  double d = t[posi + 2];
331  double tmp = d + 3.f * b;
332  double fracsq = frac * frac;
333  double fracb = frac * fracsq;
334  return fracb * (-a - 3 * c + tmp) / 6 + fracsq * ((a + c) / 2 - b) +
335  frac * (c + (-2 * a - tmp) / 6) + b;
336  } else if (posi < t.size() - 1)
337  return t[posi] + (pos - posi) * (t[posi + 1] - t[posi]);
