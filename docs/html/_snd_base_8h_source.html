240 template <typename S>
241 inline S cubic_interp(double pos, const std::vector<S> &t) {
242  size_t posi = (size_t)pos;
243  double frac = pos - posi;
244  double a = posi == 0 ? t[t.size() - 1] : t[posi - 1];
245  double b = t[posi];
246  double c = posi != t.size() - 1 ? t[posi + 1] : t[0];
247  double d =
248  posi != t.size() - 2 ? (posi != t.size() - 1 ? t[posi + 2] : t[1]) : t[0];
249  double tmp = d + 3.f * b;
250  double fracsq = frac * frac;
251  double fracb = frac * fracsq;
252  return fracb * (-a - 3 * c + tmp) / 6 + fracsq * ((a + c) / 2 - b) +
253  frac * (c + (-2 * a - tmp) / 6) + b;

275 inline S cubic_interp_lim(double pos, const std::vector<S> &t) {
276  pos = pos < 0 ? 0 : (pos < t.size() ? pos : t.size() - 1);
277  size_t posi = (size_t)pos;
278  if (posi > 0 && posi < t.size() - 2) {
279  double frac = pos - posi;
280  double a = t[posi - 1];
281  double b = t[posi];
282  double c = t[posi + 1];

284  double tmp = d + 3.f * b;

288  frac * (c + (-2 * a - tmp) / 6) + b;
289  } else if (posi < t.size() - 1)
