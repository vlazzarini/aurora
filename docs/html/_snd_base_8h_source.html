111template <typename S> class Buff : public SndBase<S> {
112 using SndBase<S>::get_sig;

117public:
122 Buff(std::size_t bsize, std::size_t vsize = def_vsize)
123 : SndBase<S>(vsize), b(bsize), wp(0), rp(0){};
124
128 std::size_t bsize() const { return b.size(); }
129
133 void bsize(std::size_t n) { b.resize(n); }
134
138 void operator()(const std::vector<S> &in) {
139 if (b.size() < vsize())
140 b.resize(vsize());
141 std::size_t end = in.size() + wp;
142 if (end < b.size()) {
143 std::copy(in.begin(), in.end(), b.begin() + wp);
144 wp = end;
145 } else {
146 std::size_t ovflw = end - b.size();
147 std::copy(in.begin(), in.end() - ovflw, b.begin() + wp);
148 std::copy(in.end() - ovflw, in.end(), b.begin());
149 wp = ovflw;
150 }
151 }
152
156 const std::vector<S> &operator()() {
157 auto &o = get_sig();
158 std::size_t end = vsize() + rp;
159 if (end < b.size()) {
160 std::copy(b.begin() + rp, b.begin() + end, o.begin());
161 rp = end;
162 } else {
163 std::size_t ovflw = end - b.size();
164 std::copy(b.begin() + rp, b.end(), o.begin());
165 std::copy(b.begin(), b.begin() + ovflw, o.end() - ovflw);
166 rp = ovflw;
167 }
168 return o;
169 }

228template <typename S> class Mix : public SndBase<S> {
229 using SndBase<S>::process;
230 using SndBase<S>::get_sig;
231
232 const std::vector<S> &mix(std::vector<S> &out, const std::vector<S> &in) {
233 std::size_t n = 0;
234 return process(
235 [&]() {
236 auto s = out[n] + in[n];
237 n++;
238 return s;
239 },
240 in.size() < this->vsize() ? in.size() : this->vsize());
241 }

275inline S linear_interp(double pos, const std::vector<S> &t) {
276 size_t posi = (size_t)pos;
277 double frac = pos - posi;
278 return t[posi] +
279 frac * ((posi != t.size() - 1 ? t[posi + 1] : t[0]) - t[posi]);
280}

287template <typename S>
288inline S cubic_interp(double pos, const std::vector<S> &t) {
289 size_t posi = (size_t)pos;
290 double frac = pos - posi;
291 double a = posi == 0 ? t[t.size() - 1] : t[posi - 1];
292 double b = t[posi];
293 double c = posi != t.size() - 1 ? t[posi + 1] : t[0];
294 double d =
295 posi != t.size() - 2 ? (posi != t.size() - 1 ? t[posi + 2] : t[1]) : t[0];
296 double tmp = d + 3.f * b;
297 double fracsq = frac * frac;
298 double fracb = frac * fracsq;
299 return fracb * (-a - 3 * c + tmp) / 6 + fracsq * ((a + c) / 2 - b) +
300 frac * (c + (-2 * a - tmp) / 6) + b;
301}

308template <typename S>
309inline S linear_interp_lim(double pos, const std::vector<S> &t) {
310 pos = pos < 0 ? 0 : (pos < t.size() ? pos : t.size() - 1);
311 size_t posi = (size_t)pos;
312 return t[posi] +
313 (posi != t.size() - 1 ? (pos - posi) * (t[posi + 1] - t[posi]) : 0);
314}

321template <typename S>
322inline S cubic_interp_lim(double pos, const std::vector<S> &t) {
323 pos = pos < 0 ? 0 : (pos < t.size() ? pos : t.size() - 1);
324 size_t posi = (size_t)pos;
325 if (posi > 0 && posi < t.size() - 2) {
326 double frac = pos - posi;
327 double a = t[posi - 1];
328 double b = t[posi];
329 double c = t[posi + 1];
330 double d = t[posi + 2];
331 double tmp = d + 3.f * b;
332 double fracsq = frac * frac;
333 double fracb = frac * fracsq;
334 return fracb * (-a - 3 * c + tmp) / 6 + fracsq * ((a + c) / 2 - b) +
335 frac * (c + (-2 * a - tmp) / 6) + b;
336 } else if (posi < t.size() - 1)
337 return t[posi] + (pos - posi) * (t[posi + 1] - t[posi]);
338 else
339 return t[posi];
340}
