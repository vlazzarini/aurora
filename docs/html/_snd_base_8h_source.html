148 template <typename S> S linear_interp(double pos, const std::vector<S> &t) {
149  size_t posi = (size_t)pos;
150  double frac = pos - posi;
151  return t[posi] +
152  frac * ((posi != t.size() - 1 ? t[posi + 1] : t[0]) - t[posi]);
