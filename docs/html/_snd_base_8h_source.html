183 template <typename S> S linear_interp(double pos, const std::vector<S> &t) {
184  size_t posi = (size_t)pos;
185  double frac = pos - posi;
186  return t[posi] +
187  frac * ((posi != t.size() - 1 ? t[posi + 1] : t[0]) - t[posi]);
188 }
