43template <typename S> class TableSet {
44 std::size_t tlen;
45 std::vector<std::vector<S>> waves;
46 S base;
47
48 void norm(std::vector<S> &wave) {
49 S max = 0.;
50 for (auto s : wave)
51 if (s > max)
52 max = s;
53 for (auto &s : wave)
54 s *= 1. / max;
55 }
56
57 void fourier(const std::vector<S> &src, S fs, int32_t type = -1) {
58 std::size_t nh;
59 uint32_t k = 0;
60 FFT<S> fft(tlen);
61 std::vector<std::complex<S>> blsp(fft.size() / 2);
62 if (type < 0) {
63 auto sp = fft.transform(src);
64 std::copy(sp, sp + blsp.size(), blsp.begin());
65 } else {
66 std::size_t n = 0;
67 for (auto &s : blsp) {
68 if (type > 1)
69 s.real(src[n++]);
70 else
71 s.imag(-src[n++]);
72 }
73 }
74 for (auto &wave : waves) {
75 double fr = base * std::pow(2, k++);
76 if (fr > fs * .375)
77 nh = 2;
78 else
79 nh = .375 * fs / fr + 1;
80 std::fill(blsp.begin() + nh, blsp.end(), std::complex<S>(0, 0));
81 auto wv = fft.transform(blsp);
82 std::copy(wv, wv + tlen, wave.begin());
83 norm(wave);
84 }
85 }
86
87 void create(S fs, uint32_t type) {
88 std::vector<S> src(tlen / 2);
89 if (type >= 0) {
90 std::size_t n = 0;
91 for (auto &s : src) {
92 switch (type) {
93 case SAW:
94 if (n)
95 s = (S)1. / n;
96 break;
97 case SQUARE:
98 if (n % 2)
99 s = (S)1. / n;
100 break;
101 case TRIANGLE:
102 if (n % 2)
103 s = (S)1. / (n * n);
104 break;
105 default:
106 s = (S)1.;
107 }
108 n++;
109 }
110 }
111 fourier(src, fs, type);
112 }
113
114 void resize(std::size_t len, S fs) {
115 tlen = len;
116 waves.resize((int)std::log2(fs / base));
117 waves.clear();
118 for (auto &w : waves)
119 w = std::vector<S>(tlen);
120 }
121
122public:
128 TableSet(uint32_t type, S fs = def_sr, std::size_t len = def_ftlen)
129 : tlen(len), waves((int)std::log2(fs / def_base), std::vector<S>(len)),
130 base((S)def_base) {
131 create(fs, type);
132 }
133
139 TableSet(const std::vector<S> &src, S b = def_base, S fs = def_sr)
140 : tlen(src.size()), waves((int)std::log2(fs / b), std::vector<S>(tlen)),
141 base(1 / (b * tlen / fs)) {
142 fourier(src, fs);
143 }
144
148 const std::vector<S> &func(S f) const {
149 int32_t num = f > base ? (int32_t)round(std::log2(f / base)) : 0;
150 return num < waves.size() ? waves[num] : waves.back();
151 }
152
158 void reset(uint32_t type, S fs, std::size_t len = def_ftlen) {
159 resize(len, fs);
160 create(std::vector<S>(tlen / 2), fs, type);
161 }
168 void reset(const std::vector<S> &src, S b, S fs) {
169 base(1 / (b * src.size() / fs));
170 resize(src.size(), fs);
171 fourier(src, fs);
172 }
173};

180template <typename S, S (*FN)(double, const std::vector<S> *) = lookupi<S>>
181class BlOsc : public Osc<S, FN> {
182 using Osc<S, FN>::ph;
183 using Osc<S, FN>::ts;
184 using Osc<S, FN>::tab;
185 const TableSet<S> *tset;
186 S ff;
187
188 S synth(S a, S f, double &phs, const std::vector<S> *t, S pm = 0) override {
189 t = ff != f ? &tset->func(f) : t;
190 return Osc<S, FN>::synth(a, f, phs, t, pm);
191 }
192
193 void reset_obj(S fs) override {
194 ff = 0;
195 Osc<S, FN>::reset_obj(fs);
196 }
197
198public:
204 BlOsc(const TableSet<S> *t, S fs = (S)def_sr, std::size_t vsize = def_vsize)
205 : Osc<S, FN>(nullptr, fs, vsize), tset(t), ff(0){};
206
210 void waveset(const TableSet<S> *t) { tset = t; }
211};
